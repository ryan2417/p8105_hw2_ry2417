import and clean the `2018 Precipitation` and `2019 Precipitation` from
`Trash-Wheel-Collection-Totals-8-6-19.xlsx`

``` r
percipitation_2018_df <-
  read_excel(
    "data/Trash-Wheel-Collection-Totals-8-6-19.xlsx",
    sheet = "2018 Precipitation",
    range = "A2:B14"
  ) %>% 
  janitor::clean_names() %>% 
  rename(percipitation_inch = total) %>% 
  filter(!is.na(percipitation_inch)) %>% 
  mutate(year = "2018")

percipitation_2019_df <-
  read_excel(
    "data/Trash-Wheel-Collection-Totals-8-6-19.xlsx",
    sheet = "2019 Precipitation",
    range = "A2:B14"
  ) %>% 
  janitor::clean_names() %>% 
  rename(percipitation_inch = total) %>% 
  filter(!is.na(percipitation_inch)) %>% 
  mutate(year = "2019")
```
